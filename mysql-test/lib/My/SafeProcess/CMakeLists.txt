  ADD_EXECUTABLE(my_safe_process safe_process_win.cc)
  ADD_EXECUTABLE(my_safe_kill safe_kill_win.cc)
  TARGET_LINK_LIBRARIES(my_safe_kill dbghelp psapi)

  INSTALL(TARGETS my_safe_kill ${INSTALL_ARGS})
ENDIF()
